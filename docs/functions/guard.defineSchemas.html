      Example

      const schemas = defineSchemas({
        User: {
          name: is.string,
          age: is.number,
          address: is.object({
            street: is.string,
            city: is.string,
            region: is.string,
          }),
        },
      });

      type User = InferSchema<typeof schemas.User>;
      // User is now { name: string; age: number; address: { street: string; city: string; region: string; } }
